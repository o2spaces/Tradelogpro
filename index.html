} /* ── SAVE / UPDATE SESSION ── */ function saveSession(d){ var sh=_sheet(SHEET.SESSIONS,HDR.SESSIONS); var portfolio=d.portfolio||'default'; var ts=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET.TRADES); var pnl=0,wins=0,losses=0,gp=0,gl=0,pnls=[]; if(ts&&ts.getLastRow()>1){ var rows=ts.getDataRange().getValues().slice(1).filter(function(r){ return r[1]===d.date && (r[2]||'default')===portfolio; }); rows.forEach(function(r){ var p=parseFloat(r[10])||0; var pf=gl?+(gp/gl).toFixed(2):0; var cum=0,peak=0,dd=0; pnls.forEach(function(p){cum+=p;if(cum>peak)peak=cum;if(peak-cum>dd)dd=peak-cum;}); var row=[ d.date||'', portfolio, d.emotion||'', d.trend||'', d.confirm||'', d.sessions||'', d.strategies||'', d.notes||'', +pnl.toFixed(2), total, wr, +gp.toFixed(2), +gl.toFixed(2), avgW, avgL, pf, +dd.toFixed(2), _now() ]; // find row matching both date AND portfolio var existing=_findRow2(sh,1,d.date,3,portfolio); if(existing>0) sh.getRange(existing,1,1,row.length).setValues([row]); else{ sh.appendRow(row); existing=sh.getLastRow(); for(var i=vals.length-1;i>=1;i--){ if(vals[i][dateCol-1]===d.date && (vals[i][portCol-1]||'default')===portfolio) sh.deleteRow(i+1); } }); return {status:'ok',date:d.date,portfolio:portfolio}; } /* ── DELETE PORTFOLIO ── */ function deletePortfolio(d){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var portfolio=String(d.portfolio||''); if(!portfolio) return {status:'error',msg:'no portfolio id'}; [SHEET.TRADES,SHEET.SESSIONS,SHEET.SUMMARY].forEach(function(name){ var sh=ss.getSheetByName(name); } }); _log('deletePortfolio','portfolio='+portfolio); return {status:'ok',portfolio:portfolio}; } /* ── RECALC SUMMARY ── */ function _recalc(date,portfolio){ if(!date) return; portfolio=portfolio||'default'; var ss=SpreadsheetApp.getActiveSpreadsheet(); var ts=ss.getSheetByName(SHEET.TRADES); var ses=ss.getSheetByName(SHEET.SESSIONS); var sum=_sheet(SHEET.SUMMARY,HDR.SUMMARY); var trades=[]; if(ts&&ts.getLastRow()>1){ trades=ts.getDataRange().getValues().slice(1).filter(function(r){ return r[1]===date && (r[2]||'default')===portfolio; }); } var pnls=trades.map(function(r){return parseFloat(r[10])||0;}); var net=pnls.reduce(function(s,v){return s+v;},0); var gp=pnls.filter(function(v){return v>0;}).reduce(function(s,v){return s+v;},0); var gl=Math.abs(pnls.filter(function(v){return v<0;}).reduce(function(s,v){return s+v;},0)); var wins=pnls.filter(function(v){return v>0;}).length; var losses=pnls.filter(function(v){return v<0;}).length; var total=trades.length; var wr=total?Math.round(wins/total*100):0; var avgW=wins?+(gp/wins).toFixed(2):0; var avgL=losses?+(gl/losses).toFixed(2):0; var pf=gl?+(gp/gl).toFixed(2):0; var best=pnls.length?Math.max.apply(null,pnls):0; var worst=pnls.length?Math.min.apply(null,pnls):0; var cum=0,peak=0,dd=0; pnls.forEach(function(p){cum+=p;if(cum>peak)peak=cum;if(peak-cum>dd)dd=peak-cum;}); var emo='',sessStr=''; if(ses&&ses.getLastRow()>1){ var sr=ses.getDataRange().getValues().slice(1).find(function(r){ return r[0]===date && (r[1]||'default')===portfolio; }); if(sr){emo=sr[2]||''; sessStr=sr[5]||'';} } var row=[ date, portfolio, +net.toFixed(2), +gp.toFixed(2), +gl.toFixed(2), total, wins, losses, wr, avgW, avgL, pf, +dd.toFixed(2), +best.toFixed(2), +worst.toFixed(2), emo, sessStr, _now() ]; var ex=_findRow2(sum,1,date,2,portfolio); if(ex>0) sum.getRange(ex,1,1,row.length).setValues([row]); else{ sum.appendRow(row); ex=sum.getLastRow(); } _colorRow(sum,ex,net>0?'#1a2e22':net<0?'#2e1a1f':'#1a1c26'); _stylePnlCell(sum,ex,3,net); } /* ── HELPERS ── */ function _sheet(name,headers){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var sh=ss.getSheetByName(name); if(!sh){sh=ss.insertSheet(name);sh.appendRow(headers);_styleHeader(sh,headers.length);} else if(sh.getLastRow()===0){sh.appendRow(headers);_styleHeader(sh,headers.length);} return sh; } function _styleHeader(sh,n){ var r=sh.getRange(1,1,1,n); r.setBackground('#0d1117').setFontColor('#4f7aff').setFontWeight('bold').setFontSize(10).setHorizontalAlignment('center'); sh.setFrozenRows(1);try{sh.autoResizeColumns(1,n);}catch(e){} } function _styleTradeRow(sh,row,result,pnl){ var bg=result==='Win'?'#1a2e22':result==='Loss'?'#2e1a1f':'#1a1c26'; _colorRow(sh,row,bg);_stylePnlCell(sh,row,11,pnl); } function _colorRow(sh,row,bg){ sh.getRange(row,1,1,sh.getLastColumn()).setBackground(bg).setFontColor('#e8eaf0').setFontSize(10); } function _stylePnlCell(sh,row,col,pnl){ var c=sh.getRange(row,col); c.setFontColor(pnl>0?'#2ecc8a':pnl<0?'#ff4d6a':'#8b90a0'); c.setFontWeight('bold'); } function _findRow(sh,col,val){ var data=sh.getDataRange().getValues(); for(var i=1;i<data.length;i++) if(String(data[i][col-1])===String(val)) return i+1; return -1; } function _findRow2(sh,col1,val1,col2,val2){ var data=sh.getDataRange().getValues(); for(var i=1;i<data.length;i++){ if(String(data[i][col1-1])===String(val1) && String(data[i][col2-1])===String(val2)) return i+1; } return -1; } function _log(action,detail){ try{ var sh=_sheet(SHEET.LOG,HDR.LOG); sh.appendRow([_now(),action,detail.substring(0,300)]); if(sh.getLastRow()>501) sh.deleteRow(2); }catch(e){} } function _now(){return Utilities.formatDate(new Date(),'Asia/Bangkok','yyyy-MM-dd HH:mm:ss');} function _json(obj){return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);} /* ── SETUP (รันครั้งแรก) ── */ function setupSheets(){ _sheet(SHEET.TRADES,HDR.TRADES); _sheet(SHEET.SESSIONS,HDR.SESSIONS); _sheet(SHEET.SUMMARY,HDR.SUMMARY); _sheet(SHEET.ASSETS,HDR.ASSETS); _sheet(SHEET.LOG,HDR.LOG); var ss=SpreadsheetApp.getActiveSpreadsheet(); var def=ss.getSheetByName('Sheet1'); if(def&&ss.getSheets().length>1) ss.deleteSheet(def); SpreadsheetApp.getUi().alert('✅ TradeLog Pro v4 พร้อม!\n\nSheets: Trades | Sessions | Summary | Assets | Log\nรองรับ Multi-Portfolio + Custom Assets + Fields ใหม่ทั้งหมด'); } function onOpen(){ SpreadsheetApp.getUi().createMenu('🔧 TradeLog Pro') .addItem('⚙️ Setup Sheets','setupSheets') .addItem('📊 Refresh All Summaries','refreshAll') .addItem('🗑 Clear Log','clearLog') .addToUi(); }function refreshAll(){ var ts=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET.TRADES); if(!ts){SpreadsheetApp.getUi().alert('ไม่พบ Sheet Trades');return;} var done={}; ts.getDataRange().getValues().slice(1).forEach(function(r){ if(r[1]){var key=r[1]+'|'+(r[2]||'default');done[key]={date:r[1],port:r[2]||'default'};} });
